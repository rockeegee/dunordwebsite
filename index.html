var hasDomain=true;

function toggleDomain(val){

  hasDomain=val;

  document.getElementById('domain-yes').className='toggle-btn'+(val?' active':'');

  document.getElementById('domain-no').className='toggle-btn'+(!val?' active':'');

  var total=399+1188+(val?100:0);

  var parts=['$399 setup'];

  if(val)parts.push('$100 domain');

  parts.push('$1,188 hosting');

  document.getElementById('calc-total').textContent='$'+total.toLocaleString();

  document.getElementById('calc-breakdown').textContent=parts.join(' + ');

}

function toggleFaq(el){

  var isOpen=el.classList.contains('open');

  document.querySelectorAll('.faq-item').forEach(function(i){i.classList.remove('open');});

  if(!isOpen)el.classList.add('open');

}
